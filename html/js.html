Input Animation JS

All it takes is adding a single class using your favorite framework, or DOM manipulation tool.

We utilize this by using some javascript to apply our input--filled class

NOTE: This is a very light setup most of this can be acomplished with a framework, or even jQuery

Vanilla JS Example: 


(function() {
// First we need to grab all of our inputs and their input fields
[].slice.call( document.querySelectorAll( 'input.input__field' ) ).forEach( function( inputEl ) {
  // In case the input is already filled..
  if( inputEl.value.trim() !== '' ) {
    inputEl.parentNode.className += ' ' + 'input--filled';
  }

  // Events:
  inputEl.addEventListener( 'focus', onInputFocus );
  inputEl.addEventListener( 'blur', onInputBlur );
} );
// For when the user clicks on our input
function onInputFocus( ev ) {
  ev.target.parentNode.className += ' ' + 'input--filled';
}
// For when the user clicks off our input
function onInputBlur( ev ) {
  // Check to make sure our input value is empty then removing our class
  if( ev.target.value.trim() === '' ) {
    var classReg = new RegExp('(^|\\s+)input--filled(\\s+|$)');
    ev.target.parentNode.className = ev.target.parentNode.className.replace(classReg, ' ');
  }
}
})();

Select Animation JS

All it takes is adding a single class using your favorite framework, or DOM manipulation tool.

We're also using SelectFx.js from Codrop Click here to see it

I'm currently working on a better way to handle this js than SelectFx.js

We utilize this by using some javascript to apply our cs-active class

(function() {
  // Select all our select elements
  [].slice.call( document.querySelectorAll( 'select.cs-select' ) ).forEach( function(el) {
    // Check to see if we are using the circle animation (if you only use a single animation this is not needed)
    if (el.className.indexOf('cs-circle') > -1) {
      // Create a new SelectFx class with some of our options (see above for selectFx Source)
      new SelectFx(el, {
        stickyPlaceholder: false,
        // When a new option is selected in our circle select box we need to trigger the image inside our main circle
        onChange: function(val){
          var img = document.createElement('img');
          img.src = '/dist/imgs/'+val+'.png';
          // Once the imge element loads in set the placeholder circles background to that image
          img.onload = function() {
            document.querySelector('.cs-circle span.cs-placeholder').style.backgroundImage = 'url(/dist/imgs/'+val+'.png)';
          };
        }
      });
    } else {
      // Else we just let SelectFx handle the element for our other types
      new SelectFx(el);
    }
  } );
})();
